        async function createTradeOffer() {
            if (myTradeItems.length === 0) return createToast("ADD ITEMS TO OFFER");
            const user = firebase.auth().currentUser;
            try {
                await db.collection("trades").add({
                    offerer: user.uid,
                    offererName: state.profile.name,
                    offererItems: myTradeItems,
                    receiverItems: [], // Items to be added by acceptor
                    status: "open",
                    offererConfirmed: true,
                    receiverConfirmed: false,
                    createdAt: firebase.firestore.FieldValue.serverTimestamp()
                });
                myTradeItems = [];
                renderTradeSlots();
                createToast("QUANTUM OFFER BROADCASTED");
                loadTradeBoard();
            } catch(e) { createToast("TRANSMISSION ERROR"); }
        }

        async function loadTradeBoard() {
            const grid = document.getElementById('partner-trade-slots');
            grid.innerHTML = '<div style="color:#0f0">Scanning network...</div>';

            const snapshot = await db.collection("trades").where("status", "==", "open").limit(10).get();
            grid.innerHTML = '';
            snapshot.forEach(doc => {
                const trade = doc.data();
                if (trade.offerer === firebase.auth().currentUser.uid) return;

                const div = document.createElement('div');
                div.style.cssText = "background:rgba(0,255,0,0.1); border:1px solid #0f0; padding:15px; margin-bottom:15px; border-radius:10px; position:relative;";
                div.innerHTML = `
                    <div style="font-size:0.7rem; color:#0f0; margin-bottom:10px;">NODE: ${trade.offererName}</div>
                    <div style="font-size:0.6rem; color:#fff; background:rgba(0,0,0,0.3); padding:8px; border-radius:4px;">
                        <b>WANTS TO GIVE:</b><br>${trade.offererItems.join(', ')}
                    </div>
                    <div style="margin-top:10px; font-size:0.5rem; color:#aaa;">Add item from your cargo to exchange:</div>
                    <button onclick="acceptTradeMutual('${doc.id}')" style="margin-top:10px; background:#0f0; color:#000; border:none; padding:10px; cursor:pointer; width:100%; font-family:'Orbitron'; font-weight:bold; border-radius:4px; transition:0.3s;">INITIALIZE EXCHANGE</button>
                `;
                grid.appendChild(div);
            });
        }

        async function acceptTradeMutual(tradeId) {
            // Step 1: Acceptor selects item to give back
            if (state.inventory.length === 0) return createToast("CARGO EMPTY");

            const overlay = document.createElement('div');
            overlay.className = 'overlay';
            overlay.style.display = 'flex';
            overlay.innerHTML = `
                <div class="menu-window" style="max-width: 400px;">
                    <h3>SELECT ITEM TO EXCHANGE</h3>
                    <div class="inv-grid" id="exchange-select-grid"></div>
                    <button onclick="this.parentElement.parentElement.remove()" style="margin-top:20px; width:100%;">CANCEL</button>
                </div>
            `;
            document.body.appendChild(overlay);

            const grid = document.getElementById('exchange-select-grid');
            state.inventory.forEach((name, i) => {
                const div = document.createElement('div');
                div.className = 'inv-item';
                div.innerHTML = `<b>${name}</b>`;
                div.onclick = async () => {
                    const myItem = state.inventory.splice(i, 1)[0];
                    overlay.remove();

                    const user = firebase.auth().currentUser;
                    const tradeRef = db.collection("trades").doc(tradeId);

                    try {
                        await db.runTransaction(async (transaction) => {
                            const doc = await transaction.get(tradeRef);
                            if (!doc.exists || doc.data().status !== "open") throw "Link Lost";

                            const tradeData = doc.data();
                            transaction.update(tradeRef, {
                                status: "completed",
                                receiver: user.uid,
                                receiverName: state.profile.name,
                                receiverItems: [myItem],
                                receiverConfirmed: true
                            });

                            // Give offerer items to receiver
                            state.inventory.push(...tradeData.offererItems);
                        });
                        createToast("QUANTUM EXCHANGE COMPLETE");
                        updateInventoryUI();
                        loadTradeBoard();
                    } catch(e) {
                        state.inventory.push(myItem); // Refund
                        createToast(e);
                    }
                };
                grid.appendChild(div);
            });
        }
